    ```typescript
    async function autoCreateIngreso(
      membershipId: string,
      monto: number,
      concepto: string,
      creadoPorId: string,
      fecha: Date
    ): Promise<void> {
      // Buscar o crear cuenta contable para ingresos MP
      const cuenta = await prisma.cuentaContable.upsert({
        where: { codigo: '401-ING' },
        create: {
          codigo: '401-ING',
          nombre: 'Ingresos por Membresías MP',
          tipo: 'INGRESO',
        },
        update: {},
      });

      await prisma.ingreso.create({
        data: {
          cuentaContableId: cuenta.id,
          concepto,
          monto,
          fecha,
          origen: 'PORTAL_MERCADOPAGO',
          referenciaId: membershipId,
          creadoPorId,
        },
      });
    }
    ```
